\def\centercell#1{%
    \ifnum#1=\z@
        ~~~~~%
    \else\ifnum#1<4 
        ~~\centercellval#1~~%
    \else\ifnum#1<7
        ~\centercellval#1~~%
    \else\ifnum#1<10
        ~\centercellval#1~%
    \else\ifnum#1<13
        \centercellval#1~%
    \else \centercellval#1%
    \fi\fi\fi\fi\fi
}

\def\displaycell#1{%
    \edef\tmpa##1{\tsetxy{\the\numexpr\xx*8+1}{\noexpand\the\numexpr\yy*4+##1}}

    \techo{\tmpa1\tcborder~~~~~~~~~\treset}%
    \techo{\tmpa2\tcborder~\tmpb~~~~~~~\tcborder~\treset}%
    \techo{\tmpa3\tcborder~\tmpb~\centercell#1~\tcborder~\treset}%
    \techo{\tmpa4\tcborder~\tmpb~~~~~~~\tcborder~\treset}%
    \techo{\tmpa5\tcborder~~~~~~~~~\treset}%
}
